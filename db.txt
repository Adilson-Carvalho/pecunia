CREATE DATABASE bd_registros;

USE bd_registros;
CREATE TABLE tb_plano_de_contas (id_conta INT NOT NULL AUTO_INCREMENT PRIMARY KEY, natureza VARCHAR(10) NOT NULL, conta VARCHAR(50) NOT NULL, sub_conta VARCHAR(50) NOT NULL);

USE bd_registros; 
CREATE TABLE tb_movimentacoes (id_registro INT NOT NULL AUTO_INCREMENT PRIMARY KEY, valor FLOAT NOT NULL, id_conta INTERGER, CONSTRAINT fk_conta FOREIGN KEY (id_conta) REFERENCES tb_plano_de_contas
 (id_conta), data DATE NOT NULL);

USE bd_registros;
CREATE TABLE tb_usuarios (id_registro INT NOT NULL AUTO_INCREMENT PRIMARY KEY, nome_usuario VARCHAR(50) NOT NULL, senha_usuario VARCHAR(20) NOT NULL);
