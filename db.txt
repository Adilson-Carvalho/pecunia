CREATE DATABASE bd_registros;
USE bd_registros;
CREATE TABLE tb_movimentacoes (id_registro INT NOT NULL AUTO_INCREMENT PRIMARY KEY, natureza VARCHAR(20) NOT NULL, valor FLOAT NOT NULL,
descr_registro VARCHAR(100) NOT NULL, data DATE NOT NULL, hora TIME NOT NULL);

USE bd_registros;
CREATE TABLE tb_usuarios (id_registro INT NOT NULL AUTO_INCREMENT PRIMARY KEY, nome_usuario VARCHAR(50) NOT NULL, senha_usuario VARCHAR(20) NOT NULL);
